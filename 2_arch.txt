// When reinstalling Arch:
# 1. Set Samsung screen: blue light to 0%; red 60%; green 55%

# 2. Install Arch from [arch-install.txt] script + ALWAYS restore files using 
#    same backup command (rsync)

# 3. make: highlighter

# 4. nautilus-typeahead: (download it from AUR as the community version 
#       3.34.0-1 with sudo pacman -S nautilus-typeahead is broken):
	nautilus-typeahead  3.34.2-1 
	# install gnome-terminal-transparency then reboot for terminal opening
	# view Modified date below files; not size (from settings)

# 5. enable scrollbars: add this line to ~/.config/gtk-3.0/settings.ini
	gtk-overlay-scrolling=0
	gtk-primary-button-warps-slider=1

# 6. set notifications settings:
	mkdir ~/.config/dunst && cp /usr/share/dunst/dunstrc ~/.config/dunst

# 7. sudo vim /boot/loader/loader.conf # reduce grub bootloader countdown to 1; 
      # this is for [systemd-boot] bootloader (shows 2 options in mid screen).
      # for GRUB: sudo vim /boot/grub/grub.cfg # look for timeout. see:
      https://bbs.archlinux.org/viewtopic.php?id=251950

# 8. update calculator icon (all this is not needed anymore):
sudo rm -f /usr/share/icons/hicolor/symbolic/apps/org.gnome.Calculator-symbolic.svg
sudo cp /usr/share/icons/hicolor/scalable/apps/org.gnome.Calculator.svg
        /usr/share/icons/hicolor/symbolic/apps/org.gnome.Calculator-symbolic.svg
# change ones you don't like eg:
# sudo rm /usr/share/icons/breeze/apps/48/okular.svg
# sudo rm /usr/share/icons/breeze-dark/apps/48/okular.svg
# sudo cp firefox.png /usr/share/icons/elementary-xfce/apps/128/

sudo rm /usr/share/icons/hicolor/256x256/apps/chromium.png
sudo rm /usr/share/icons/hicolor/128x128/apps/chromium.png
sudo rm /usr/share/icons/hicolor/64x64/apps/chromium.png
sudo rm /usr/share/icons/hicolor/48x48/apps/chromium.png
sudo rm /usr/share/icons/hicolor/32x32/apps/chromium.png
sudo rm /usr/share/icons/hicolor/24x24/apps/chromium.png
sudo rm /usr/share/icons/hicolor/16x16/apps/chromium.png

# 9. Uninstall:
pip uninstall pip # remove it from all existing conda environments
sudo pacman -R snapd # don't be enslaved/hooked/addicted to updates or 3rd party companies.

# 10. to install a desktop (allow desktop icons) on awesome:
sudo pacman -S pcmanfm
