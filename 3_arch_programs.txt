// When reinstalling Arch:
# 1. Set Samsung screen: blue light to 0%; red 60%; green 55%

# 2. Install Arch from [1_arch_install.txt] script

# 3. ALWAYS restore files using same backup command (rsync)

# 4. make: highlighter
    
# 5. enable scrollbars: add this line to ~/.config/gtk-3.0/settings.ini
	gtk-overlay-scrolling=0
	gtk-primary-button-warps-slider=1

# 6. set notifications settings:
	mkdir ~/.config/dunst && cp /usr/share/dunst/dunstrc ~/.config/dunst

# 7. sudo vim /boot/loader/loader.conf # reduce grub bootloader countdown to 1; 
      # this is for [systemd-boot] bootloader (shows 2 options in mid screen).
      # for GRUB: sudo vim /boot/grub/grub.cfg # look for timeout. see:
      https://bbs.archlinux.org/viewtopic.php?id=251950

# 8. Uninstall:
pip uninstall pip # remove it from all existing conda environments
sudo pacman -R snapd # don't be enslaved/hooked/addicted to updates or 3rd party companies.

# 9. to install a desktop (allow desktop icons) on awesome:
sudo pacman -S pcmanfm
